# ADI icon workflow parity — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the lost `embed-icons.sh` toolchain with a stdlib Python generator that regenerates both `lib/adi/adi-components.d2` and `lib/sw/sw-components.d2` from source SVG files and keeps the `ADI_COMPONENTS` / `SW_COMPONENTS` lists in `d2/__init__.py` in sync. One-shot extraction script recovers the 64 existing ADI icons from base64. CI check prevents drift.

**Architecture:** Two scripts. `scripts/extract_adi_icons.py` runs once to decode each base64 blob in the current `adi-components.d2` into `lib/adi/icons/*.svg`, then invokes the generator in check-mode to self-verify. `scripts/embed_icons.py` is the ongoing tool: it reads `lib/<lib>/icons/*.svg`, re-encodes them, writes the `.d2` file, and rewrites the `*_COMPONENTS` list in `d2/__init__.py` between `# BEGIN/END AUTOGEN:<lib>` marker comments. A `nox -s embed_check` session (new CI workflow) runs the generator in `--check` mode to fail PRs that forget to regenerate.

**Tech Stack:** Python 3.10+ stdlib only (`base64`, `pathlib`, `argparse`, `re`, `difflib`, `sys`). pytest. nox. No new runtime deps.

**Spec:** `docs/superpowers/specs/2026-04-19-adi-icon-workflow-parity-design.md`.

```python
"""Regenerate lib/<lib>/<lib>-components.d2 and the matching
*_COMPONENTS list in d2/__init__.py from source SVG icons.

Usage:
    python scripts/embed_icons.py --library {adi,sw,all} [--check]

In --check mode nothing is written; the script exits 0 if the
on-disk files already match what would be generated, 1 otherwise
with a unified diff printed to stderr.
"""

from __future__ import annotations

import argparse
import base64
import difflib
import re
import sys
from dataclasses import dataclass
from pathlib import Path

_ROOT = Path(__file__).resolve().parent.parent

# Seeded from d2/__init__.py's ADI_COMPONENTS at spec time. When a new ADI
# icon is added (Sub-project 2), append its class name here explicitly.
_ADI_ORDER: list[str] = [
    "adc", "dac", "amplifier", "filter-lp", "filter-bp", "filter-hp", "pll",
    "mixer", "oscillator", "comparator", "summer", "multiplexer", "switch",
    "voltage-reference", "voltage-regulator", "sensor", "dsp-fpga", "clock",
    "driver", "isolator", "attenuator", "beamformer", "axi-interconnect",
    "spi", "gpio", "i2c", "aurora", "pfd", "charge-pump", "loop-filter",
    "vco", "divider", "ldo", "dc-dc-buck", "dc-dc-boost", "pmic",
    "power-monitor", "hot-swap", "battery-charger", "lna", "pa", "gain-block",
    "upconverter", "downconverter", "rf-switch", "rf-transceiver", "dds",
    "rs485", "can", "ethernet-phy", "lvds", "jesd204", "inamp", "tia", "vga",
    "current-sense", "clock-buffer", "jitter-cleaner", "imu", "temp-sensor",
    "afe", "motor-driver", "gate-driver", "digipot",
]


@dataclass
class GenerateResult:
    in_sync: bool
    message: str = ""


def _render_d2(lib_name: str, order: list[str], icons_dir: Path) -> str:
    lines = [
        "# Auto-generated by scripts/embed_icons.py — do not edit manually",
        f"# Regenerate with: python scripts/embed_icons.py --library {lib_name}",
        "",
        "classes: {",
    ]
    for name in order:
        svg = (icons_dir / f"{name}.svg").read_bytes()
        b64 = base64.b64encode(svg).decode("ascii")
        lines.append(f"  {name}: {{")
        lines.append(f'    icon: "data:image/svg+xml;base64,{b64}"')
        lines.append("  }")
    lines.append("}")
    lines.append("")  # trailing newline
    return "\n".join(lines)


def _render_list(list_name: str, order: list[str]) -> str:
    body = [f"{list_name} = ["]
    for name in order:
        body.append(f'    "{name}",')
    body.append("]")
    return "\n".join(body)


_MARKER_RE = re.compile(
    r"(# BEGIN AUTOGEN:(?P<lib>[a-z]+)[^\n]*\n)(.*?)(# END AUTOGEN:(?P=lib)\n)",
    re.DOTALL,
)


def _rewrite_init_block(init_text: str, lib_name: str, new_body: str) -> str:
    def _repl(match: re.Match[str]) -> str:
        if match.group("lib") != lib_name:
            return match.group(0)
        # Preserve the original list-description comment (the line between BEGIN marker
        # and the list declaration): find it inside the current block, keep it as-is.
        original_body = match.group(3)
        comment_match = re.match(
            r"(# [A-Z][^\n]*\n)",
            original_body.lstrip("\n"),
        )
        description = comment_match.group(1) if comment_match else ""
        return f"{match.group(1)}{description}{new_body}\n{match.group(4)}"

    replaced, n = _MARKER_RE.subn(_repl, init_text)
    if n == 0:
        raise ValueError(f"no BEGIN AUTOGEN:{lib_name} marker found in init file")
    return replaced


def generate_library(
    *,
    lib_name: str,
    list_name: str,
    order: list[str],
    icons_dir: Path,
    d2_output: Path,
    init_py: Path,
    check: bool,
) -> GenerateResult:
    # Validate sources first
    missing = [n for n in order if not (icons_dir / f"{n}.svg").is_file()]
    if missing:
        return GenerateResult(False, f"missing icons in {icons_dir}: {missing}")
    on_disk = {p.stem for p in icons_dir.glob("*.svg")}
    orphans = sorted(on_disk - set(order))
    if orphans:
        return GenerateResult(
            False,
            f"unknown icons in {icons_dir}: {orphans} — add to _ORDER['{lib_name}']",
        )

    new_d2 = _render_d2(lib_name, order, icons_dir)
    new_list = _render_list(list_name, order)

    current_init = init_py.read_text()
    new_init = _rewrite_init_block(current_init, lib_name, new_list)

    if check:
        d2_diff = _diff(d2_output, new_d2)
        init_diff = _diff(init_py, new_init)
        if d2_diff or init_diff:
            return GenerateResult(False, d2_diff + init_diff)
        return GenerateResult(True)
    d2_output.write_text(new_d2)
    init_py.write_text(new_init)
    return GenerateResult(True)


def _diff(existing: Path, generated: str) -> str:
    current = existing.read_text() if existing.exists() else ""
    if current == generated:
        return ""
    return "".join(
        difflib.unified_diff(
            current.splitlines(keepends=True),
            generated.splitlines(keepends=True),
            fromfile=f"a/{existing.name}",
            tofile=f"b/{existing.name} (generated)",
        )
    )


def _main(argv: list[str] | None = None) -> int:
    parser = argparse.ArgumentParser(description=__doc__)
    parser.add_argument("--library", choices=["adi", "sw", "all"], required=True)
    parser.add_argument("--check", action="store_true")
    args = parser.parse_args(argv)

    libs = ["adi", "sw"] if args.library == "all" else [args.library]
    exit_code = 0
    for lib in libs:
        if lib == "adi":
            order = _ADI_ORDER
            list_name = "ADI_COMPONENTS"
        else:
            order = sorted(
                p.stem for p in (_ROOT / "lib" / "sw" / "icons").glob("*.svg")
            )
            list_name = "SW_COMPONENTS"
        result = generate_library(
            lib_name=lib,
            list_name=list_name,
            order=order,
            icons_dir=_ROOT / "lib" / lib / "icons",
            d2_output=_ROOT / "lib" / lib / f"{lib}-components.d2",
            init_py=_ROOT / "d2" / "__init__.py",
            check=args.check,
        )
        if not result.in_sync:
            print(result.message, file=sys.stderr)
            exit_code = 1
    return exit_code


if __name__ == "__main__":
    sys.exit(_main())
```

```python
"""One-shot migration: decode the base64 blobs in lib/adi/adi-components.d2
into source SVG files under lib/adi/icons/, then verify the new generator
round-trips losslessly.

Run once:

    python scripts/extract_adi_icons.py

After a successful run, commit lib/adi/icons/*.svg and the updated header
in lib/adi/adi-components.d2.
"""

from __future__ import annotations

import base64
import re
import subprocess
import sys
from pathlib import Path

_ROOT = Path(__file__).resolve().parent.parent
_ADI_D2 = _ROOT / "lib" / "adi" / "adi-components.d2"
_ICONS_DIR = _ROOT / "lib" / "adi" / "icons"

_ENTRY_RE = re.compile(
    r'^\s{2}(?P<name>[a-z][a-z0-9-]*):\s*\{\s*\n'
    r'^\s{4}icon:\s*"data:image/svg\+xml;base64,(?P<b64>[A-Za-z0-9+/=]+)"\s*\n'
    r'^\s{2}\}\s*\n',
    re.MULTILINE,
)


def _parse_entries(d2_text: str) -> dict[str, str]:
    return {m.group("name"): m.group("b64") for m in _ENTRY_RE.finditer(d2_text)}


def main() -> int:
    before = _parse_entries(_ADI_D2.read_text())
    if not before:
        print(f"ERROR: no icon entries found in {_ADI_D2}", file=sys.stderr)
        return 1

    _ICONS_DIR.mkdir(parents=True, exist_ok=True)
    for name, b64 in before.items():
        (_ICONS_DIR / f"{name}.svg").write_bytes(base64.b64decode(b64))
    print(f"Extracted {len(before)} icons to {_ICONS_DIR}")

    # Run the generator in write mode so the header updates to reference
    # the new script.
    result = subprocess.run(
        [sys.executable, "scripts/embed_icons.py", "--library", "adi"],
        cwd=_ROOT,
    )
    if result.returncode != 0:
        print("ERROR: embed_icons.py failed during regeneration", file=sys.stderr)
        return 1

    # Verify body base64 content is byte-identical (header may differ).
    after = _parse_entries(_ADI_D2.read_text())
    missing = set(before) - set(after)
    extra = set(after) - set(before)
    differ = {k for k in before if k in after and before[k] != after[k]}
    if missing or extra or differ:
        print(
            "ERROR: round-trip failed — "
            f"missing={sorted(missing)} extra={sorted(extra)} differ={sorted(differ)}",
            file=sys.stderr,
        )
        return 1

    print("Regen round-trips: OK.")
    return 0


if __name__ == "__main__":
    sys.exit(main())
```
